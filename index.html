<script>


function reserve(){


const date = document.getElementById("date").value;

const time = document.getElementById("time").value;

const user = document.getElementById("user").value;


if(!date || !time || !user){

alert("Please select date, time, and name");

return;

}


const url = "https://script.google.com/macros/s/AKfycbypaF4kzybUo9w0B81mnEWBWmnFLF1oGIZ9mIaUOdNSh-S8-iDBwmA1IWthpYyXRv_d/exec";


const data = {

date: date,

time: time,

user: user

};


fetch(url,{

method:"POST",

body: JSON.stringify(data)

})

.then(response => response.text())

.then(result => {

alert("Reservation complete");

})


.catch(error => {

alert("Error sending reservation");

console.error(error);

});


}
